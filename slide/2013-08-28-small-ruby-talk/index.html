Singleton Pattern


require 'singleton'
class Mario
  include Singleton


  def greets
    puts "It's me, Mario!"
  end
end
Mario.instance.greets

Singleton Pattern Underneath


# (simplified)
class Mario
  def self.instance
    @instance ||= new
  end
  def greets
    puts "It's me, Mario!"
  end
end
Mario.instance.greets

Singleton Pattern *Simplest*



Mario = Object.new



  def Mario.greets
    puts "It's me, Mario!"
  end
